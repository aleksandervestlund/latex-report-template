    \section{File Structure}

    This template assumes the following file layout:

    \dirtree{%
        .1 ..
        .2 main.tex\DTcomment{Your main file}.
        .2 packages.sty\DTcomment{Contains package imports and commands}.
        .2 references.bib\DTcomment{Your bibliography file}.
        .2 glossary.tex\DTcomment{Your glossary file}.
        .2 Makefile\DTcomment{To run \mintinline{bash}{pdflatex}}.
        .2 README.md\DTcomment{For info on GitHub}.
        .2 Sections\DTcomment{Folder for text}.
        .3 00\_Title\DTcomment{Title page}.
        .3 01\_Frontmatter\DTcomment{Front page}.
        .3 99\_Appendix\DTcomment{Appendix}.
        .2 Images\DTcomment{Folder for images}.
        .3 MIT\_logo.png\DTcomment{\mintinline{latex}{\mitlogo}}.
        .3 NTNU\_logo\_EN.png\DTcomment{\mintinline{latex}{\ntnulogo} (language$\in$\{american,  british\})}.
        .3 NTNU\_logo\_NB.png\DTcomment{\mintinline{latex}{\ntnulogo} (language=norsk)}.
        .3 SNU\_logo.png\DTcomment{\mintinline{latex}{\snulogo}}.
        .2 Code\DTcomment{Folder for code snippets}.
    }
\end{appendices}
